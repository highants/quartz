    <script>
        const display = document.getElementById('display');
        const buttons = document.querySelector('.buttons');

        let currentInput = '0'; // 現在の入力値
        let firstOperand = null; // 最初のオペランド
        let operator = null; // 選択された演算子
        let waitingForSecondOperand = false; // 2番目のオペランド入力待ちフラグ

        // 表示を更新する関数
        function updateDisplay() {
            display.textContent = currentInput;
            // 表示桁数制限 (オプション)
            // const maxDigits = 12;
            // if (display.textContent.length > maxDigits) {
            //     display.textContent = parseFloat(display.textContent).toExponential(maxDigits - 5); // 指数表記に
            // }
        }

        // 数値入力処理
        function inputDigit(digit) {
            if (currentInput === '0' || waitingForSecondOperand) {
                currentInput = digit;
                waitingForSecondOperand = false;
            } else {
                // 表示桁数制限を簡易的に実装
                if (currentInput.length >= 15) return;
                currentInput += digit;
            }
        }

        // 小数点入力処理
        function inputDecimal() {
            if (waitingForSecondOperand) { // 演算子後の最初の入力が '.' の場合
                currentInput = '0.';
                waitingForSecondOperand = false;
                return;
            }
            if (!currentInput.includes('.')) {
                currentInput += '.';
            }
        }

        // 演算子処理
        function handleOperator(nextOperator) {
             // "%" の処理を追加 (単項演算として簡易的に実装)
             if (nextOperator === '%') {
                if (currentInput === '0') return;
                currentInput = (parseFloat(currentInput) / 100).toString();
                updateDisplay();
                // % の後は連続計算をしないようにリセットする (単純化のため)
                // waitingForSecondOperand = true; // または計算状態をクリア
                return; // 他の演算子処理に進まない
            }


            const inputValue = parseFloat(currentInput);

            // エラー状態なら演算子を受け付けない
            if (currentInput === 'Error') return;

            // 最初の演算子入力、または連続して演算子が押された場合
            if (operator && waitingForSecondOperand) {
                operator = nextOperator; // 最後の演算子を優先
                return;
            }

            // 最初のオペランドがまだない場合は、現在の入力を最初のオペランドとする
            if (firstOperand === null && !isNaN(inputValue)) {
                firstOperand = inputValue;
            } else if (operator) { // 演算子があり、2番目の入力がある場合 (例: 1 + 2 + の場合)
                const result = performCalculation();
                 if (result === 'Error') {
                     currentInput = 'Error';
                     firstOperand = null;
                     operator = null;
                     waitingForSecondOperand = false;
                     updateDisplay();
                     return; // エラー時は以降の処理を中断
                 }
                currentInput = String(result);
                firstOperand = result;
                updateDisplay(); // 中間結果を表示
            }

            waitingForSecondOperand = true;
            operator = nextOperator;
        }

        // 計算実行関数 (内部用)
        function performCalculation() {
            if (operator === null || waitingForSecondOperand || firstOperand === null) {
                 // console.log("Calculation skipped: ", operator, waitingForSecondOperand, firstOperand);
                return parseFloat(currentInput); // '=' を連続で押した場合など
            }

            const secondOperand = parseFloat(currentInput);
            let result = 0;

            if (isNaN(firstOperand) || isNaN(secondOperand)) {
                return 'Error'; // 不正な数値の場合
            }

            switch (operator) {
                case '+':
                    result = firstOperand + secondOperand;
                    break;
                case '-':
                    result = firstOperand - secondOperand;
                    break;
                case '*':
                    result = firstOperand * secondOperand;
                    break;
                case '/':
                    if (secondOperand === 0) {
                        return 'Error'; // ゼロ除算エラー
                    }
                    result = firstOperand / secondOperand;
                    break;
                default:
                    // console.log("Unknown operator:", operator);
                    return secondOperand; // 演算子がない場合は現在の数値を返す
            }

            // 結果が扱える範囲か簡易チェック (もっと厳密なチェックも可能)
            if (!isFinite(result)) {
                 return 'Error'; // オーバーフロー/アンダーフロー
            }


             // 浮動小数点数の精度問題を少し緩和 (表示桁数を制限する方がベター)
             // result = parseFloat(result.toPrecision(12));


            return result;
        }


        // '=' ボタンの処理
        function calculate() {
            if (operator === null || waitingForSecondOperand || currentInput === 'Error') {
                // 演算子がない、または演算子を押した直後、またはエラー状態なら何もしない
                return;
            }
            const result = performCalculation();

            if (result === 'Error') {
                currentInput = 'Error';
                 // エラー後も計算継続を可能にするなら以下のリセットは不要
                 operator = null;
                 firstOperand = null;
                 waitingForSecondOperand = false;

            } else {
                 currentInput = String(result);
                 // 計算後、結果を次の計算の最初のオペランドにする準備
                 firstOperand = result; // '='連続押しでの再計算や、結果に続けて演算する場合
                 operator = null; // '=' を押したら演算子は一旦クリア
                 waitingForSecondOperand = true; // 次は数字入力を期待する (仕様によるが一般的)

                  // 計算結果表示後、数字が押されたら新しい入力が始まるようにする
                  // waitingForSecondOperand = true のため、inputDigit内で処理される
            }
             updateDisplay();

             // // 計算後に状態をクリアして完全に新規入力から始める場合:
             // updateDisplay();
             // firstOperand = null; // または firstOperand = result; で計算継続
             // operator = null;
             // waitingForSecondOperand = false; // AC に近い状態 (仕様次第)

        }


        // AC (All Clear) 処理
        function clearAll() {
            currentInput = '0';
            firstOperand = null;
            operator = null;
            waitingForSecondOperand = false;
            updateDisplay();
        }

        // CE (Clear Entry) 処理
        function clearEntry() {
             // もしエラーが表示されていたらACと同じ動作にする
            if (currentInput === 'Error') {
                 clearAll();
                 return;
             }

            // 演算子入力後 (waitingForSecondOperand = true) のCEは意味がない、
            // または表示を0に戻すが内部状態は変えないなど仕様による。
            // ここでは単純に現在の入力を'0'にする。
            currentInput = '0';
            // もし演算子入力直後 (waiting状態) にCEが押された場合の挙動は検討が必要。
            // 例えば、第二オペランド入力をキャンセルしたいなら、`waitingForSecondOperand`も調整が必要かも。
            // 現状ではシンプルに表示を0にするだけに留める。
             waitingForSecondOperand = false; // CE後は通常入力に戻す
             updateDisplay();
        }


        // イベントリスナー (イベント委任)
        buttons.addEventListener('click', (event) => {
            const { target } = event;
            const { action, value } = target.dataset;

            if (!action) return; // ボタン以外がクリックされた場合は無視

            switch (action) {
                case 'number':
                    inputDigit(value);
                    updateDisplay();
                    break;
                case 'decimal':
                    inputDecimal();
                    updateDisplay();
                    break;
                case 'operator':
                     handleOperator(value);
                    // 演算子を押したタイミングでの表示更新は任意
                    // updateDisplay(); // handleOperator内で更新される場合もある
                    break;
                case 'calculate':
                    calculate();
                     // updateDisplay(); // calculate内で更新される
                    break;
                case 'clear':
                    clearAll();
                    break;
                case 'clear-entry':
                     clearEntry();
                     break;
                 default:
                     console.log("Unknown action:", action);
            }

            // デバッグ用に状態を表示
            // console.log('State:', { currentInput, firstOperand, operator, waitingForSecondOperand });
        });

        // 初期表示
        updateDisplay();

    </script>
